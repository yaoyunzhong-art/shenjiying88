### 8.2 静默执行脚本

```bash
#!/bin/bash
# testing-silent-run.sh - 后台静默测试执行
# 执行方式: nohup ./testing-silent-run.sh &

LOG_FILE="/var/log/testing-silent-$(date +%Y%m%d).log"
ERROR_FILE="/var/log/testing-error-$(date +%Y%m%d).log"

echo "[$(date)] Testing silent run started" >> $LOG_FILE

# 7:00 冒烟测试
7:00 {
    echo "[$(date)] Running smoke test" >> $LOG_FILE
    cd /path/to/tests && npm run test:smoke >> $LOG_FILE 2>> $ERROR_FILE
}

# 8:00-14:00 回归测试
8:00-14:00 {
    echo "[$(date)] Running regression test" >> $LOG_FILE
    cd /path/to/tests && npm run test:regression >> $LOG_FILE 2>> $ERROR_FILE
}

# 9:00 安全扫描
9:00 {
    echo "[$(date)] Running security scan" >> $LOG_FILE
    /path/to/security/scan.sh >> $LOG_FILE 2>> $ERROR_FILE
}

# 10:00-12:00 压力测试
10:00-12:00 {
    echo "[$(date)] Running stress test" >> $LOG_FILE
    /path/to/loadtest/run.sh >> $LOG_FILE 2>> $ERROR_FILE
}

echo "[$(date)] Testing run completed" >> $LOG_FILE
```
